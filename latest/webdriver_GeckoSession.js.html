'use strict';

const Session = require('./Session');
const AddonBuilder = require('../addon/AddonBuilder');

/**
 * GeckoSession represents an session of the geckodriver.
 *
 * @extends Session
 *
 * @example
 * let driver = new GeckoDriver();
 * let session = await GeckoSession.create(driver);
 *
 * // navigates to github.com
 * await session.navigateTo('https://github.com');
 *
 * // closes current session
 * await session.close();
 */
class GeckoSession extends Session {

  /**
   * Creates a new session of the WebDriver via geckodriver and return a
   * GeckoSession.
   *
   * @param {WebDriver} driver The WebDriver instance.
   * @param {object=} capabilities The {@link
   * https://firefox-source-docs.mozilla.org/testing/geckodriver/Capabilities.html|Firefox
   * capabilities} of the geckodriver session.
   * @returns GeckoSession
   */
  static async create(driver, capabilities) {
    let sessionId = await driver.newSession(capabilities);
    return new GeckoSession(sessionId, driver);
  }

  /**
   * Installs temporary addon from the path.
   *
   * @param {string} path The path to addon directory containing <code>manifest.json</code>.
   */
  async installAddonFromPath(path) {
    let data = await new AddonBuilder(path).build();
    return this.installAddon(data);
  }

  /**
   * Installs temporary addon from the path.
   *
   * @param {Buffer} data in zip format.
   */
  async installAddon(data) {
    await this.driver.installAddon(this.sessionId, data, true);
  }
}

module.exports = GeckoSession;

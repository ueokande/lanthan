'use strict';

const Element = require('./Element');

/**
 * Session is an base class for the WebDriver.
 */
class Session {

  /**
   * @param {string} sessionId The session ID.
   * @param {WebDriver} driver The client of the WebDriver.
   */
  constructor(sessionId, driver) {
    this.sessionId = sessionId;
    this.driver = driver;
  }

  /**
   * Closes current session of the WebDriver.
   */
  async close() {
    await this.driver.deleteSession(this.sessionId);
  }

  /**
   * Retrieves element from the current context by CSS selector.
   *
   * @param {string} css The CSS selector.
   * @return {Element} The retrieved element.
   */
  async findElementByCSS(css) {
    let id = await this.driver.findElement(this.sessionId, 'css selector', css);
    return new Element(this.driver, this.sessionId, id);
  }

  /**
   * Retrieves elements from the current context by CSS selector.
   *
   * @param {string} css The CSS selector.
   * @return {Element} The retrieved elements.
   */
  async findElementsByCSS(css) {
    let ids = await this.driver.findElements(
      this.sessionId, 'css selector', css);
    return ids.map(id => new Element(this.driver, this.sessionId, id));
  }

  /**
   * Navigates to the page on the current session.
   *
   * @param {string} url The URL to be navigated.
   */
  async navigateTo(url) {
    await this.driver.navigateTo(this.sessionId, url);
  }

  /**
   * Create new window.
   *
   * @async
   * @param {string=} url
   * @param {type=} [type="tab"] The open type ("window" or "tab", default to "tab")
   */
  newWindow(url = undefined, type = 'tab') {
    return this.driver.newWindow(this.sessionId, url, type);
  }

  /**
   * Select the current window.
   *
   * @async
   * @param {string=} url
   * @param {type=} [type="tab"] The open type ("window" or "tab", default to "tab")
   */
  switchToWindow(handle) {
    return this.driver.switchToWindow(this.sessionId, handle);
  }

  /**
   * Get all window handles.
   *
   * @async
   * @return {string[]} The window handles.
   */
  getWindowHandles() {
    return this.driver.getWindowHandles(this.sessionId);
  }

  /**
   * Get current window handle.
   *
   * @async
   * @returns {string} The current window handle.
   */
  getWindowHandle() {
    return this.driver.getWindowHandle(this.sessionId);
  }

  /**
   * Select the current frame in the window.
   *
   * @async
   */
  switchToFrame(id) {
    return this.driver.switchToFrame(this.sessionId, id);
  }

  /**
   * Select the top frame in the window.
   *
   * @async
   */
  switchToParentFrame() {
    return this.driver.switchToParentFrame(this.sessionId);
  }

  /**
   * Execute scripts in the current context and returns the executed result.
   * The script can be string, or JavaScript function.
   *
   * @example
   * // execute script by string
   * await session.executeScript(`document.querySelector('#description').value`)
   *
   * // execute scripts by function
   * await session.executeScript(() => document.querySelector('#description').value)
   * @async
   * @param {string|function} script The script or function to be executed.
   * @return {object} The result of the executed script.
   */
  executeScript(script) {
    return this.driver.executeScript(this.sessionId, script);
  }
}

module.exports = Session;

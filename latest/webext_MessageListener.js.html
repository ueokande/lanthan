'use strict';

/**
 * MessageListener receives an message from the output stream.
 *
 * @example
 * let listener = new MessageListener(stdout);
 * listener.onMessage((msg) => console.info(msg));
 * listener.onError((err) => console.info(err));
 * listener.listen();
 */
class MessageListener {

  /**
   * @param {Stream} r The output stream to listen messages.
   */
  constructor(r) {
    this.r = r;
    this.onMessageListener = () => {};
    this.onErrorListener = () => {};
  }

  /**
   * Registers a message handler.
   *
   * @param {function} listener A message listener.
   */
  onMessage(listener) {
    this.onMessageListener = listener;
  }

  /**
   * Registers an error handler.
   *
   * @param {function} listener A message listener.
   */
  onError(listener) {
    this.onErrorListener = listener;
  }

  /**
   * Starts to listen message from the stream.
   */
  listen() {
    let buffer = Buffer.alloc(0);
    let remaining = 4;
    let readingHeader = true;

    this.r.on('readable', () => {
      try {
        let chunk = null;
        while ((chunk = this.r.read(remaining)) !== null) {
          remaining -= chunk.length;
          buffer = Buffer.concat([buffer, chunk]);

          if (remaining !== 0) {
            continue;
          }

          if (readingHeader) {
            remaining = buffer.readUInt32LE();
          } else {
            remaining = 4;
            let message = JSON.parse(buffer);
            this.onMessageListener(message);
          }
          readingHeader = !readingHeader;
          buffer = Buffer.alloc(0);
        }
      } catch (e) {
        this.onErrorListener(e);
      }
    });
  }
}

module.exports = MessageListener;

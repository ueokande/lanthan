'use strict';

const request = require('request-promise-native');
const WebDriver = require('./WebDriver');

/**
 * GeckoDriver is a client of the {@link
 * https://firefox-source-docs.mozilla.org/testing/geckodriver/index.html#|geckodriver}.
 *
 * @extends WebDriver
 *
 * @example
 * let driver = new GeckoDriver();
 * let sessionId = await driver.newSession();
 *
 * // Switch to the first window
 * await driver.switchToWindow(sessionId, handles[0]);
 */
class GeckoDriver extends WebDriver {

  /**
   * @param {number} [port=4444] A port of the WebDriver server.
   * @param {string} [address="127.0.0.1"] An address of the WebDriver server.
   */
  constructor(port = 4444, address = '127.0.0.1') {
    super(port, address);
  }

  /**
   * Installs an addon in ZIP archive.
   *
   * @example
   * let driver = new GeckoDriver();
   * let sessionId = await driver.newSession();
   *
   * // Install an addon from zip file.
   * await driver.installAddon(sessionId, fs.readFileSync('path/to/addon.zip'));
   *
   * @param {string} sessionId The session ID.
   * @param {Buffer} data The addon data in ZIP archive.
   * @param {boolean} [temporary=false] True if the installed addon is temporary.
   */
  async installAddon(sessionId, data, temporary = false) {
    await request({
      url: `${this.endpoint}/session/${sessionId}/moz/addon/install`,
      method: 'POST',
      json: {
        addon: data.toString('base64'),
        temporary,
      },
    });
  }
}

module.exports = GeckoDriver;

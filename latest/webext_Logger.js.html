'use strict';

const fs = require('fs');

/**
 * An logger class to output file.
 *
 * @example
 * let logger = new Logger('server.log');
 *
 * logger.info('starting server...');
 * logger.warn('something occurs', err);
 * logger.error('an error occurs', err);
 */
class Logger {

  /**
   * @param {string} path The path to output log file.
   */
  constructor(path) {
    this.path = path;
  }

  /**
   * Outputs logs with INFO level.
   *
   * @param {...object} messages The messages to logged.
   */
  info(...messages) {
    this.append('INFO', ...messages);
  }

  /**
   * Outputs logs with WARN level.
   *
   * @param {...object} messages The messages to logged.
   */
  warn(...messages) {
    this.append('WARN', ...messages);
  }

  /**
   * Outputs logs with ERROR level.
   *
   * @param {...object} messages The messages to logged.
   */
  error(...messages) {
    this.append('ERROR', ...messages);
  }

  /**
   * Outputs messages with log level to the file.
   *
   * @private
   * @param {string} level The log level.
   * @param {...object} messages The messages to logged.
   */
  append(level, ...messages) {
    let line = `${new Date().toISOString()} [${level}] ${messages.join(' ')}\n`;
    fs.appendFileSync(this.path, line);
  }
}

module.exports = Logger;

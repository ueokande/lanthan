'use strict';

/**
 * Enum for keys for {@link Element#sendKeys}.
 *
 * @example
 * // send Ctrl+T
 * await element.sendKeys(Ket.Control, 't');
 *
 * @readonly
 * @enum {string}
 */
const Keys = {

  /** Return key */
  Return: '\uE006',

  /** Enter key */
  Enter: '\uE007',

  /** Shift key */
  Shift: '\uE008',

  /** Control key */
  Control: '\uE009',

  /** Alt key */
  Alt: '\uE00A',

  /** Escape key */
  Escape: '\uE00C',

  /** Tab key */
  Tab: '\uE004',
};


module.exports = Keys;

'use strict';

/**
 * Element represents an element in the WebDriver.
 *
 * @example
 * let session = await GeckoSession.create(new GeckoDriver())
 * let elem = await session.findElement('body');
 *
 * await elem.sendKeys('a');  // send key event
 * await elem.lick();         // send click event
 */
class Element {

  /**
   * @param {WebDriver} driver The client of the {@link WebDriver}
   * @param {string} sessionId The session ID.
   * @param {object} elementId The element ID
   */
  constructor(driver, sessionId, elementId) {
    this.driver = driver;
    this.sessionId = sessionId;
    this.elementId = elementId;
  }

  /**
   * Retrieves the attribute value of the element.
   *
   * @async
   * @param {string} name The name of the attribute.
   * @returns {string} The attribute value of the element.
   */
  getAttribute(name) {
    return this.driver.getElementAttribute(
      this.sessionId, Object.values(this.elementId)[0], name,
    );
  }

  /**
   * Retrieves the CSS value of the element.
   *
   * @async
   * @param {string} name The name of the style.
   * @returns {string} The style value of the element.
   */
  getStyle(name) {
    return this.driver.getElementStyle(
      this.sessionId, Object.values(this.elementId)[0], name,
    );
  }

  /**
   * Retrieves the text content of the element.
   *
   * @async
   * @returns {string} Text content of the element.
   */
  getText() {
    return this.driver.getElementText(
      this.sessionId, Object.values(this.elementId)[0],
    );
  }

  /**
   * Retrieves the tag name of the element.
   *
   * @async
   * @returns {string} The tag name of the element.
   */
  getTagName() {
    return this.driver.getElementTagName(
      this.sessionId, Object.values(this.elementId)[0],
    );
  }

  /**
   * Sends key events to the element.
   *
   * @example
   * let input = await session.findElement('input');
   * await input.sendKeys('hello')
   * @example
   * const { Key } = require('lanthan');
   * let body = await session.findElement('body');
   * // Send Ctrl+T
   * await body.sendKeys(Key.Ctrl, 't');
   *
   * @async
   * @param {...string} keys The key sequence to be sent.
   */
  sendKeys(...keys) {
    return this.driver.elementSendKeys(
      this.sessionId, Object.values(this.elementId)[0], ...keys
    );
  }

  /**
   * Sends click event to the element.
   *
   * @example
   * let button = await session.findElement('button');
   * await button.click()
   * @async
   */
  click() {
    return this.driver.elementClick(
      this.sessionId, Object.values(this.elementId)[0],
    );
  }

  /**
   * Check if the element is displayed.
   *
   * @async
   * @returns {boolean} <code>true</code> if the element is displayed.
   */
  isDisplayed() {
    return this.driver.isElementDisplayed(
      this.sessionId, Object.values(this.elementId)[0],
    );
  }

  /**
   * Check if the element is selected.
   *
   * @async
   * @returns {boolean} <code>true</code> if the element is selected.
   */
  isSelected() {
    return this.driver.isElementSelected(
      this.sessionId, Object.values(this.elementId)[0],
    );
  }
}

module.exports = Element;
